#include <iostream>

int main() {
    int n;
    std::cin >> n;

    for (int x = -n; x <= n; x++) { 
        for (int y = -n; y <= n; y++) { 
            if (std::abs(x) + std::abs(y) <= n) { 
                std::cout << "* ";
            } else {
                std::cout << ". ";
            }
        }
        std::cout << std::endl; 
    }

    return 0;
}
